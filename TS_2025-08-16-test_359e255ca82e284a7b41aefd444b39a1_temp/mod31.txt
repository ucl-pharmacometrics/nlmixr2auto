f <- function(){
ini({
lvc <- 0.223
lvmax <- 0.056
lkm <- 4.788
eta.vmax ~ 0.1
sigma_add <-5.0434
sigma_prop <-0.2
})
model({
vc = exp(lvc)
vmax = exp(lvmax+eta.vmax)
km = exp(lkm)
d/dt(central) <- - (vmax / (km + central / vc)) / vc * central
cp <- central/vc
cp ~ prop(sigma_prop) + add(sigma_add)
})
}
